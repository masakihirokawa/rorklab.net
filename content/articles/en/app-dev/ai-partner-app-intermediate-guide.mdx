# Polished AI Companion Apps: UI, Memory & Personas

You've built your first chat app. Now let's make it production-ready. We'll add multiple AI personas, persistent memory, smooth animations, and push notifications—just like Replika or Character.ai.

---

## Architecture Overview

An intermediate companion app has several layers:

1. **UI Layer** — Polished chat bubbles, typing indicators, avatars
2. **Storage Layer** — SQLite for structured conversation history
3. **Memory System** — Conversation summaries injected into prompts
4. **Persona System** — Switch between different AI characters
5. **Notifications** — Remind users to chat (with permission)

Let's build each piece.

## Setting Up SQLite

First, install and initialize SQLite:

```bash
npx expo install expo-sqlite
```

Create a database service:

```typescript
// services/database.ts
import * as SQLite from 'expo-sqlite';

const db = SQLite.openDatabase('companion.db');

export interface StoredMessage {
  id: string;
  conversationId: string;
  role: 'user' | 'assistant';
  content: string;
  timestamp: number;
}

export interface Conversation {
  id: string;
  personaId: string;
  title: string;
  createdAt: number;
  lastMessageAt: number;
}

export async function initializeDatabase() {
  return new Promise<void>((resolve, reject) => {
    db.transaction((tx) => {
      tx.executeSql(
        `CREATE TABLE IF NOT EXISTS conversations (
          id TEXT PRIMARY KEY,
          personaId TEXT NOT NULL,
          title TEXT NOT NULL,
          createdAt NUMBER NOT NULL,
          lastMessageAt NUMBER NOT NULL
        );`,
        [],
        () => console.log('Conversations table created'),
        (_, error) => reject(error)
      );

      tx.executeSql(
        `CREATE TABLE IF NOT EXISTS messages (
          id TEXT PRIMARY KEY,
          conversationId TEXT NOT NULL,
          role TEXT NOT NULL,
          content TEXT NOT NULL,
          timestamp NUMBER NOT NULL,
          FOREIGN KEY (conversationId) REFERENCES conversations(id)
        );`,
        [],
        () => console.log('Messages table created'),
        (_, error) => reject(error)
      );

      tx.executeSql(
        `CREATE TABLE IF NOT EXISTS personas (
          id TEXT PRIMARY KEY,
          name TEXT NOT NULL,
          systemPrompt TEXT NOT NULL,
          avatar TEXT,
          color TEXT
        );`,
        [],
        () => {
          console.log('Personas table created');
          resolve();
        },
        (_, error) => reject(error)
      );
    });
  });
}

export async function saveMessage(
  message: StoredMessage
): Promise<void> {
  return new Promise((resolve, reject) => {
    db.transaction((tx) => {
      tx.executeSql(
        `INSERT INTO messages (id, conversationId, role, content, timestamp)
         VALUES (?, ?, ?, ?, ?);`,
        [
          message.id,
          message.conversationId,
          message.role,
          message.content,
          message.timestamp,
        ],
        () => resolve(),
        (_, error) => reject(error)
      );
    });
  });
}

export async function getConversationMessages(
  conversationId: string
): Promise<StoredMessage[]> {
  return new Promise((resolve, reject) => {
    db.transaction((tx) => {
      tx.executeSql(
        `SELECT * FROM messages WHERE conversationId = ? ORDER BY timestamp ASC;`,
        [conversationId],
        (_, result) => {
          resolve(result.rows._array as StoredMessage[]);
        },
        (_, error) => reject(error)
      );
    });
  });
}
```

## Memory System with Summaries

Implement conversation memory by summarizing old messages:

```typescript
// services/memory.ts
import axios from 'axios';

const CLAUDE_API_KEY = process.env.CLAUDE_API_KEY;

export async function summarizeConversation(
  messages: Array<{ role: string; content: string }>
): Promise<string> {
  const messagesToSummarize = messages.slice(0, -10);

  if (messagesToSummarize.length < 5) {
    return '';
  }

  try {
    const response = await axios.post(
      'https://api.anthropic.com/v1/messages',
      {
        model: 'claude-3-5-sonnet-20241022',
        max_tokens: 500,
        messages: [
          {
            role: 'user',
            content: `Summarize this conversation in 2-3 sentences, focusing on
what you learned about the user:\n\n${messagesToSummarize
              .map((m) => `${m.role}: ${m.content}`)
              .join('\n')}`,
          },
        ],
      },
      {
        headers: {
          'Content-Type': 'application/json',
          'x-api-key': CLAUDE_API_KEY,
          'anthropic-version': '2023-06-01',
        },
      }
    );

    return response.data.content[0].text;
  } catch (error) {
    console.error('Summarization failed:', error);
    return '';
  }
}

export async function getAIResponseWithMemory(
  messages: Array<{ role: string; content: string }>,
  persona: any,
  memorySummary: string
): Promise<string> {
  const memoryContext = memorySummary
    ? `\n\nPrevious conversation summary: ${memorySummary}`
    : '';

  const systemPrompt = persona.systemPrompt + memoryContext;

  try {
    const response = await axios.post(
      'https://api.anthropic.com/v1/messages',
      {
        model: 'claude-3-5-sonnet-20241022',
        max_tokens: 1024,
        system: systemPrompt,
        messages: messages,
      },
      {
        headers: {
          'Content-Type': 'application/json',
          'x-api-key': CLAUDE_API_KEY,
          'anthropic-version': '2023-06-01',
        },
      }
    );

    return response.data.content[0].text;
  } catch (error) {
    console.error('API Error:', error);
    throw error;
  }
}
```

## Polished Chat UI with Typing Indicators

Create a refined chat component:

```typescript
// components/ChatBubble.tsx
import React from 'react';
import { View, Text, StyleSheet } from 'react-native';

interface Props {
  role: 'user' | 'assistant';
  content: string;
  isLoading?: boolean;
  personaColor?: string;
}

export default function ChatBubble({
  role,
  content,
  isLoading,
  personaColor = '#f0f0f0',
}: Props) {
  const isUser = role === 'user';

  return (
    <View style={[styles.container, isUser && styles.userContainer]}>
      <View
        style={[
          styles.bubble,
          isUser ? styles.userBubble : { backgroundColor: personaColor },
        ]}
      >
        <Text style={[styles.text, isUser && styles.userText]}>
          {isLoading ? '✓ Typing...' : content}
        </Text>
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    marginVertical: 6,
    flexDirection: 'row',
    justifyContent: 'flex-start',
  },
  userContainer: {
    justifyContent: 'flex-end',
  },
  bubble: {
    maxWidth: '80%',
    paddingHorizontal: 14,
    paddingVertical: 10,
    borderRadius: 16,
  },
  userBubble: {
    backgroundColor: '#007AFF',
  },
  text: {
    fontSize: 15,
    color: '#333',
    lineHeight: 20,
  },
  userText: {
    color: '#fff',
  },
});
```

## Persona Switching Screen

Allow users to switch between AI companions:

```typescript
// screens/PersonaScreen.tsx
import React from 'react';
import { View, TouchableOpacity, Text, StyleSheet, ScrollView } from 'react-native';
import { PERSONAS } from '../services/personas';

interface Props {
  onSelectPersona: (personaId: string) => void;
  currentPersonaId: string;
}

export default function PersonaScreen({
  onSelectPersona,
  currentPersonaId,
}: Props) {
  return (
    <ScrollView style={styles.container}>
      <Text style={styles.title}>Choose Your Companion</Text>

      {PERSONAS.map((persona) => (
        <TouchableOpacity
          key={persona.id}
          style={[
            styles.personaCard,
            currentPersonaId === persona.id && styles.selectedCard,
          ]}
          onPress={() => onSelectPersona(persona.id)}
        >
          <Text style={styles.avatarEmoji}>{persona.avatarEmoji}</Text>
          <View style={styles.personaInfo}>
            <Text style={styles.personaName}>{persona.name}</Text>
            <Text style={styles.personaDesc}>{persona.description}</Text>
          </View>
          {currentPersonaId === persona.id && (
            <Text style={styles.checkmark}>✓</Text>
          )}
        </TouchableOpacity>
      ))}
    </ScrollView>
  );
}
```

## Push Notifications for Daily Check-ins

Add gentle reminders:

```bash
npx expo install expo-notifications
```

```typescript
// services/notifications.ts
import * as Notifications from 'expo-notifications';

export async function requestNotificationPermission() {
  const { status } = await Notifications.requestPermissionsAsync();
  return status === 'granted';
}

export async function scheduleDailyNotification() {
  await Notifications.scheduleNotificationAsync({
    content: {
      title: 'Your companion is thinking of you!',
      body: 'Come chat and catch up',
      sound: 'default',
    },
    trigger: {
      hour: 10,
      minute: 0,
      repeats: true,
    },
  });
}
```

## Dark Mode Support

Add theme support using Context:

```typescript
// context/ThemeContext.tsx
import React, { createContext, useState } from 'react';

export interface Theme {
  dark: boolean;
  colors: {
    background: string;
    text: string;
    border: string;
    userBubble: string;
  };
}

const LIGHT_THEME: Theme = {
  dark: false,
  colors: {
    background: '#fff',
    text: '#000',
    border: '#eee',
    userBubble: '#007AFF',
  },
};

const DARK_THEME: Theme = {
  dark: true,
  colors: {
    background: '#1a1a1a',
    text: '#fff',
    border: '#333',
    userBubble: '#0084FF',
  },
};

export const ThemeContext = createContext(LIGHT_THEME);

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<Theme>(LIGHT_THEME);

  const toggleTheme = () => {
    setTheme(theme.dark ? LIGHT_THEME : DARK_THEME);
  };

  return (
    <ThemeContext.Provider value={theme}>
      {children}
    </ThemeContext.Provider>
  );
}
```
