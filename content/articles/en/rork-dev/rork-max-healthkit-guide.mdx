## HealthKit Authorization Setup

Every HealthKit app must request permission before reading health data. Here's the pattern Rork Max generates:

### Info.plist Keys

```xml
<key>NSHealthShareUsageDescription</key>
<string>We use HealthKit to display your fitness metrics and activity data.</string>
<key>NSHealthUpdateUsageDescription</key>
<string>We write workout data to HealthKit to track your exercise sessions.</string>
```

### Authorization Manager (SwiftUI + async/await)

```swift
import HealthKit
import SwiftUI

@MainActor
class HealthKitManager: ObservableObject {
    let healthStore = HKHealthStore()

    @Published var isAuthorized = false
    @Published var stepCount: Double = 0
    @Published var heartRate: Double = 0
    @Published var sleepHours: Double = 0
    @Published var weeklySteps: [DailyStepData] = []

    private let typesToRead: Set<HKObjectType> = [
        HKQuantityType.quantityType(forIdentifier: .stepCount)!,
        HKQuantityType.quantityType(forIdentifier: .heartRate)!,
        HKCategoryType.categoryType(forIdentifier: .sleepAnalysis)!
    ]

    func requestAuthorization() async {
        guard HKHealthStore.isHealthDataAvailable() else { return }

        do {
            try await healthStore.requestAuthorization(
                toShare: [],
                read: typesToRead
            )
            isAuthorized = true
            await fetchAllHealthData()
        } catch {
            print("HealthKit authorization failed: \(error.localizedDescription)")
        }
    }

    func fetchAllHealthData() async {
        await fetchTodayStepCount()
        await fetchHeartRate()
        await fetchSleepData()
        await fetchWeeklySteps()
    }
}
```

## Querying Step Count Data

Rork Max consistently generates clean `HKStatisticsQuery` patterns for cumulative data like steps:

```swift
func fetchTodayStepCount() async {
    guard let stepType = HKQuantityType.quantityType(forIdentifier: .stepCount) else { return }

    let startOfDay = Calendar.current.startOfDay(for: Date())
    let predicate = HKQuery.predicateForSamples(
        withStart: startOfDay,
        end: Date(),
        options: .strictStartDate
    )

    let query = HKStatisticsQuery(
        quantityType: stepType,
        quantitySamplePredicate: predicate,
        options: .cumulativeSum
    ) { [weak self] _, result, _ in
        guard let sum = result?.sumQuantity() else { return }
        Task { @MainActor in
            self?.stepCount = sum.doubleValue(for: .count())
        }
    }

    healthStore.execute(query)
}
```

The `@MainActor` isolation here is a Swift 6 concurrency pattern—Rork Max generates this correctly without any manual prompting.

## Weekly Steps with Apple Charts Framework

Visualizing historical data requires querying multiple days and rendering them with the Charts framework. Here's how Rork Max handles it:

```swift
struct DailyStepData: Identifiable {
    let id = UUID()
    let date: Date
    let steps: Double

    var dayLabel: String {
        date.formatted(.dateTime.weekday(.abbreviated))
    }
}

func fetchWeeklySteps() async {
    guard let stepType = HKQuantityType.quantityType(forIdentifier: .stepCount) else { return }

    var components = DateComponents()
    components.day = -6
    let startDate = Calendar.current.date(byAdding: components, to: Calendar.current.startOfDay(for: Date()))!

    let predicate = HKQuery.predicateForSamples(withStart: startDate, end: Date())

    let query = HKStatisticsCollectionQuery(
        quantityType: stepType,
        quantitySamplePredicate: predicate,
        options: .cumulativeSum,
        anchorDate: startDate,
        intervalComponents: DateComponents(day: 1)
    )

    query.initialResultsHandler = { [weak self] _, result, _ in
        var data: [DailyStepData] = []
        result?.enumerateStatistics(from: startDate, to: Date()) { stats, _ in
            let steps = stats.sumQuantity()?.doubleValue(for: .count()) ?? 0
            data.append(DailyStepData(date: stats.startDate, steps: steps))
        }
        Task { @MainActor in
            self?.weeklySteps = data
        }
    }

    healthStore.execute(query)
}
```

## Main Dashboard with Ring Progress

The circular progress ring—inspired by Apple Watch Activity rings—creates an immediately familiar UI:

```swift
struct HealthDashboardView: View {
    @StateObject private var healthManager = HealthKitManager()
    let stepGoal: Double = 10_000

    var stepProgress: Double {
        min(healthManager.stepCount / stepGoal, 1.0)
    }

    var body: some View {
        NavigationStack {
            ScrollView {
                VStack(spacing: 24) {
                    // Circular progress ring
                    ZStack {
                        Circle()
                            .stroke(Color.blue.opacity(0.15), lineWidth: 22)

                        Circle()
                            .trim(from: 0, to: stepProgress)
                            .stroke(
                                LinearGradient(
                                    colors: [.blue, .cyan, .mint],
                                    startPoint: .topLeading,
                                    endPoint: .bottomTrailing
                                ),
                                style: StrokeStyle(lineWidth: 22, lineCap: .round)

                            .animation(.spring(duration: 1.2), value: stepProgress)

                        VStack(spacing: 2) {
                            Text(healthManager.stepCount, format: .number)
                                .font(.system(size: 38, weight: .bold, design: .rounded))
                            Text("of \(Int(stepGoal)) steps")
                                .font(.caption)
                                .foregroundStyle(.secondary)

                    }
                    .frame(width: 210, height: 210)
                    .padding(.top, 8)

                    // Metrics grid
                    LazyVGrid(
                        columns: [GridItem(.flexible()), GridItem(.flexible())],
                        spacing: 14

                            icon: "heart.fill",
                            title: "Heart Rate",

                            color: .red

                            icon: "bed.double.fill",

                }
                .padding(.vertical)
            }
            .navigationTitle("Health Today")
        }
        .task {
            await healthManager.requestAuthorization()
        }
    }
}
```

## Key Considerations for App Store Approval

HealthKit apps go through additional App Store review scrutiny. Keep these in mind when building with Rork Max:

**Usage descriptions must be specific.** Apple rejects vague strings like "We need health access." Be explicit: describe which data types you read and why. Rork Max generally generates good defaults, but review them before submission.

**HealthKit availability check is mandatory.** Always include `HKHealthStore.isHealthDataAvailable()` before accessing the store. Rork Max includes this check automatically—never remove it, as HealthKit isn't available on all devices.

**Health data stays on device.** If you're transmitting any health data to a server, this must be clearly disclosed. Building with Rork Max in a local-only mode (no backend sync) simplifies your privacy story considerably.

**Real device testing is essential.** iOS Simulator has limited HealthKit support. Budget time to test on actual iPhone and Apple Watch hardware before App Store submission.
