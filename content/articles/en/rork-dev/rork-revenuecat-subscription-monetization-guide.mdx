1. Dashboard → Configuration → Apple App Store
2. Enter your Bundle ID (e.g., `com.example.myapp`)
3. Generate **Shared Secret** in App Store Connect:
   - App Store Connect → Your App → App Information → App Store Server

#### Connecting Google Play

1. Dashboard → Configuration → Google Play
2. Go to Google Play Console
3. Create service account and download JSON file
4. Upload JSON to RevenueCAT

### Step 3: Get Your API Key

1. Account → API Keys
2. Generate New API Key
3. Copy Public Key (you'll need this in Rork code)

Store it as an environment variable:


### Step 2: Create Service Class

```typescript
// src/services/revenuecat.ts
import Purchases from '@react-native-purchases/purchases';

const API_KEY = process.env.EXPO_PUBLIC_REVENUECAT_KEY || '';
const ENTITLEMENT_ID = 'premium_access'; // Define in RevenueCAT dashboard

export interface SubscriptionStatus {
  isPremium: boolean;
  expirationDate?: string;
  autoRenewEnabled?: boolean;
  productId?: string;
}

export class RevenueCATService {
  static async initialize() {
    try {
      await Purchases.configure({
        apiKey: API_KEY,
        appUserID: null, // Auto-generates unique ID
      });
    } catch (error) {
      console.error('RevenueCAT init failed:', error);
    }
  }

  static async getOfferings() {
    try {
      const offerings = await Purchases.getOfferings();
      return offerings.current?.availablePackages || [];
    } catch (error) {
      console.error('Failed to fetch offerings:', error);
      return [];
    }
  }

  static async purchasePackage(pkg: any) {
    try {
      const { customerInfo } = await Purchases.purchasePackage(pkg);
      return { success: true, customerInfo };
    } catch (error: any) {
      if (error.code === 'PurchaseCancelledError') {
        return { success: false, cancelled: true };
      }
      return { success: false, error: error.message };
    }
  }

  static async restorePurchases() {
    try {
      const customerInfo = await Purchases.restorePurchases();
      return customerInfo;
    } catch (error) {
      console.error('Restore failed:', error);
      return null;
    }
  }

  static async getSubscriptionStatus(): Promise<SubscriptionStatus> {
    try {
      const customerInfo = await Purchases.getCustomerInfo();
      const entitlement = customerInfo.entitlements.active[ENTITLEMENT_ID];

      if (entitlement) {
        return {
          isPremium: true,
          expirationDate: entitlement.expirationDate,
          autoRenewEnabled: entitlement.renewalInfo?.willRenew,
          productId: entitlement.productIdentifier,
        };
      }
      return { isPremium: false };
    } catch (error) {
      console.error('Status check failed:', error);
      return { isPremium: false };
    }
  }
}
```

### Step 3: Global State Management

```typescript
// src/context/SubscriptionContext.tsx
import React, { createContext, useEffect, useState } from 'react';
import { RevenueCATService } from '../services/revenuecat';

interface SubscriptionContextValue {
  isPremium: boolean;
  offerings: any[];
  isLoading: boolean;
  purchase: (pkg: any) => Promise<void>;
  restore: () => Promise<void>;
}

export const SubscriptionContext = createContext<SubscriptionContextValue | undefined>();

export function SubscriptionProvider({ children }: { children: React.ReactNode }) {
  const [isPremium, setIsPremium] = useState(false);
  const [offerings, setOfferings] = useState([]);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    initSubs();
  }, []);

  const initSubs = async () => {
    try {
      await RevenueCATService.initialize();
      const status = await RevenueCATService.getSubscriptionStatus();
      setIsPremium(status.isPremium);

      const pkgs = await RevenueCATService.getOfferings();
      setOfferings(pkgs);
    } catch (error) {
      console.error('Init failed:', error);
    } finally {
      setIsLoading(false);
    }
  };

  const purchase = async (pkg: any) => {
    try {
      setIsLoading(true);
      const result = await RevenueCATService.purchasePackage(pkg);

      if (result.success) {
        const status = await RevenueCATService.getSubscriptionStatus();
        setIsPremium(status.isPremium);
      }
    } finally {
      setIsLoading(false);
    }
  };

  const restore = async () => {
    try {
      setIsLoading(true);
      await RevenueCATService.restorePurchases();
      const status = await RevenueCATService.getSubscriptionStatus();
      setIsPremium(status.isPremium);
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <SubscriptionContext.Provider
      value={{ isPremium, offerings, isLoading, purchase, restore }}
    >
      {children}
    </SubscriptionContext.Provider>
  );
}

export function useSubscription() {
  const context = React.useContext(SubscriptionContext);
  if (!context) {
    throw new Error('useSubscription must be inside SubscriptionProvider');
  }
  return context;
}
```

```typescript
// src/components/Paywall.tsx
import React, { useState } from 'react';
import { View, TouchableOpacity, Text, ActivityIndicator } from 'react-native';
import RevenueCatUI from '@react-native-purchases/purchases-ui';

export function StandardPaywall() {
  const [loading, setLoading] = useState(false);

  const show = async () => {
    try {
      setLoading(true);
      await RevenueCatUI.presentPaywallIfNeeded({
        requiredEntitlementIdentifier: 'premium_access',

    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return <ActivityIndicator size="large" />;

  return (
    <TouchableOpacity
      onPress={show}
      style={{

      }}
    >
      <Text style={{ color: 'white', textAlign: 'center', fontSize: 16 }}>
        Unlock Premium

    </TouchableOpacity>
  );
}
```

## App Store and Google Play Review

### What Apple Checks

- Are purchases clearly labeled?
- Can users restore purchases on new devices?
- Is subscription auto-renewal explained?
- How easy is it to cancel?

**Pass these by**: Clear copy explaining everything, RestorePurchases button, cancel link in app settings.

### What Google Play Checks

- Is BillingClient used correctly?
- Are test accounts whitelisted?
- Do purchases actually work?

Less strict than Apple, but still important.

### Common Rejections

- "Subscription terms not explained" → Add clear copy to paywall
- "No restore purchase button" → Add it to Settings
- "Cancel method not obvious" → Link to subscription management in app

## FAQ

### Q: Do I need my own backend?

**A**: No. RevenueCAT handles everything. If you want to verify receipts server-side later, use RevenueCAT's /customers API.

### Q: Can I have multiple subscription tiers?

**A**: Yes. Create multiple products (`premium_monthly`, `premium_annual`) with the same entitlement ID. Users can switch between them.

### Q: What about one-time purchases?

**A**: RevenueCAT supports Consumable (one-time) purchases too. Use the same API.

### Q: How do I handle regional pricing?

**A**: App Store Connect and Google Play let you set per-region prices. RevenueCAT automatically shows the right price.

### Q: Does Family Sharing work?

**A**: On iOS, yes — Apple handles it automatically. On Android, you need to configure family library separately.
