### Integrating Gemini API: Code Example

Example code for calling Gemini API from Rork backend (Node.js / Python):

```javascript
// Rork Backend (Node.js / Firebase Cloud Functions)

const axios = require('axios');

async function getAiRecommendation(userId, dailyIntake, goal) {
  // 1. Fetch user's historical data from database
  const userData = await db.collection('users').doc(userId).get();

  // 2. Build Gemini API request
  const prompt = `
User has the following information:
- Daily nutrition intake: ${dailyIntake.calories} kcal, ${dailyIntake.protein}g protein
- Goal: ${goal === 'weight_loss' ? 'weight loss' : goal === 'muscle_gain' ? 'muscle gain' : 'maintenance'}
- Average activity over past 7 days: ${userData.avgSteps} steps

Based on this, suggest optimal nutrition for their next meal.
Output format:
{
  "recommendation": "Suggestion text (1-2 sentences)",
  "suggested_foods": ["food1", "food2", "food3"],
  "estimated_calories": 400,
  "estimated_protein": 20
}
  `;

  // 3. Google Cloud authentication (service account)
  const authResponse = await axios.post(
    'https://oauth2.googleapis.com/token',
    {
      client_email: process.env.GCP_SERVICE_ACCOUNT_EMAIL,
      private_key: process.env.GCP_PRIVATE_KEY,
      type: 'service_account'
    }
  );

  // 4. Call Gemini API (prototype implementation)
  // Note: Gemini API implementation may vary as of March 2026
  const geminiResponse = await axios.post(
    'https://generativelanguage.googleapis.com/v1/models/gemini-3.1-pro:generateContent',
    {
      contents: [{
        parts: [{
          text: prompt
        }]
      }]
    },
    {
      headers: {
        'Authorization': `Bearer ${authResponse.data.access_token}`,
        'Content-Type': 'application/json'
      }
    }
  );

  // 5. Parse response JSON
  const content = geminiResponse.data.candidates[0].content.parts[0].text;
  const recommendation = JSON.parse(content);

  // 6. Save recommendation to database (audit log & user history)
  await db.collection('recommendations').add({
    user_id: userId,
    created_at: new Date(),
    recommendation_text: recommendation.recommendation,
    suggested_foods: recommendation.suggested_foods,
    estimated_calories: recommendation.estimated_calories
  });

  return recommendation;
}

// Rork Backend endpoint
exports.getRecommendation = async (req, res) => {
  const { user_id, daily_intake, goal } = req.body;

  try {
    const recommendation = await getAiRecommendation(user_id, daily_intake, goal);
    res.json({ success: true, data: recommendation });
  } catch (error) {
    console.error('Gemini API Error:', error);
    res.status(500).json({ success: false, error: error.message });
  }
};
```

### Integrating Google Fit and Apple Health

```javascript
// Fetch health data from Google Fit

async function fetchGoogleFitData(userId, accessToken) {
  // Google Fit REST API
  const response = await axios.post(
    'https://www.googleapis.com/fitness/v1/users/me/dataset:aggregate',
    {
      aggregateBy: [
        {
          dataTypeName: 'com.google.step_count.delta',
          dataSourceId: 'derived:com.google.step_count.delta:com.google.android.gms:estimated_steps'
        },
        {
          dataTypeName: 'com.google.calories.expended',
          dataSourceId: 'derived:com.google.calories.expended:com.google.android.gms:from_activities'
        }
      ],
      bucketByTime: { durationMillis: 86400000 }, // 1-day buckets
      startTimeMillis: Date.now() - (7 * 24 * 60 * 60 * 1000), // Past 7 days
      endTimeMillis: Date.now()
    },
    {
      headers: {
        'Authorization': `Bearer ${accessToken}`
      }
    }
  );

  // Format data
  const dailyData = response.data.bucket.map(bucket => ({
    date: new Date(parseInt(bucket.startTimeMillis)),
    steps: bucket.dataset[0].point[0]?.value[0]?.intVal || 0,
    caloriesBurned: bucket.dataset[1].point[0]?.value[0]?.fpVal || 0
  }));

  return dailyData;
}

// Fetch data from Apple Health (iOS)
// Using rn-health library for React Native

import Health from 'rn-health';

async function fetchAppleHealthData(userId) {
  const permissions = {

      ]
    }
  };

  Health.initHealthKit(permissions, (err, result) => {
    if (err) console.log('HealthKit Auth Error:', err);

    const startDate = new Date();
    startDate.setDate(startDate.getDate() - 7);

    Health.getStepCount(
      { startDate, endDate: new Date(), period: 'day' },
      (err, results) => {
        if (err) return;
        // Save results to Firestore
      }
    );
  });
}
```
