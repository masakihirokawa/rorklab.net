# Rork Max AI Features Guide

Integrating AI into Rork Max apps transforms user experience and creates competitive advantages. This guide explores two complementary approaches: on-device machine learning (TensorFlow Lite) and cloud-based AI APIs (OpenAI, Gemini), with practical implementation patterns for each.

## On-Device ML vs Cloud AI Decision Matrix

| Factor | On-Device ML | Cloud AI API |
|--------|--------------|--------------|
| Latency | Low (<100ms) | Higher (300ms+) |
| Privacy | ✓ Local only | ⚠️ Server transmission |
| Bundle Size | Large (10–100MB) | Small |
| Accuracy | Limited | High |
| Updates | Manual download | Automatic |
| Cost | Development | Per-API-call |
| Connectivity | Not required | Required |
| Model Control | Full | Limited |

<Callout type="info">
**Hybrid Strategy**: Use on-device ML for privacy-sensitive, low-latency tasks (face detection, text analysis). Use cloud APIs for high-accuracy requirements (natural language understanding, content generation). Combine both for optimal UX.
</Callout>

### Load Pre-trained Models

**src/utils/tflite.ts**
```typescript
import * as tf from '@tensorflow/tfjs';
import '@tensorflow/tfjs-react-native';

export async function loadModel(modelUrl: string) {
  try {
    // Use WASM backend for better performance
    await tf.setBackend('wasm');
    const model = await tf.loadLayersModel(modelUrl);
    console.log('Model loaded:', model.summary());
    return model;
  } catch (error) {
    console.error('Failed to load model:', error);
    throw error;
  }
}

export async function predictImage(
  model: tf.LayersModel,
  imageData: number[][]
): Promise<number[]> {
  const input = tf.tensor(imageData, [1, 224, 224, 3]);

  try {
    const prediction = model.predict(input) as tf.Tensor;
    const result = await prediction.data();
    return Array.from(result);
  } finally {
    tf.dispose([input]);
  }
}

export async function preprocessImage(
  base64: string
): Promise<number[][]> {
  // Convert base64 to tensor and normalize to [0, 1]
  const img = tf.browser.fromPixels(
    decodeBase64ToImageData(base64),
    3
  );
  const resized = tf.image.resizeBilinear(img, [224, 224]);
  const normalized = resized.div(tf.scalar(255));
  const array = await normalized.array();

  tf.dispose([img, resized, normalized]);
  return array as number[][];
}

function decodeBase64ToImageData(base64: string): ImageData {
  // Implementation: decode base64 to ImageData
  const binary = atob(base64);
  const bytes = new Uint8Array(binary.length);
  for (let i = 0; i < binary.length; i++) {
    bytes[i] = binary.charCodeAt(i);
  }
  return new ImageData(bytes, 224, 224);
}
```

## Part 2: Text Embeddings & Semantic Search

Implement on-device text embeddings for semantic search without cloud dependencies.

**src/utils/embeddings.ts**
```typescript
import * as tf from '@tensorflow/tfjs';

interface Embedding {
  text: string;
  vector: number[];
}

export function cosineSimilarity(a: number[], b: number[]): number {
  const dotProduct = a.reduce((sum, val, i) => sum + val * b[i], 0);
  const normA = Math.sqrt(a.reduce((sum, val) => sum + val * val, 0));
  const normB = Math.sqrt(b.reduce((sum, val) => sum + val * val, 0));

  if (normA === 0 || normB === 0) return 0;
  return dotProduct / (normA * normB);
}

export async function getTextEmbedding(
  text: string,
  model: tf.LayersModel
): Promise<number[]> {
  // Tokenize text (simplified; use a proper tokenizer for production)
  const tokens = text
    .toLowerCase()
    .split(/\W+/)
    .filter(Boolean)
    .map((word) => word.charCodeAt(0)); // Simple character codes

  const input = tf.tensor2d([tokens], [1, tokens.length]);

  try {
    const output = model.predict(input) as tf.Tensor;
    const data = await output.data();
    return Array.from(data);
  } finally {
    tf.dispose([input]);
  }
}

export function semanticSearch(
  query: string,
  queryEmbedding: number[],
  documents: Embedding[]
): Embedding[] {
  const scored = documents.map((doc) => ({
    ...doc,
    similarity: cosineSimilarity(queryEmbedding, doc.vector),
  }));

  return scored
    .sort((a, b) => b.similarity - a.similarity)
    .slice(0, 5);
}
```

## Part 3: OpenAI API Integration

### GPT-4o Chat Integration

**src/services/openaiService.ts**
```typescript
export interface ChatMessage {
  role: 'user' | 'assistant' | 'system';
  content: string;
}

export class OpenAIService {
  private baseURL = 'https://api.openai.com/v1';

  constructor(private apiKey: string) {}

  async chat(
    messages: ChatMessage[],
    model = 'gpt-4o'
  ): Promise<string> {
    try {
      const response = await fetch(
        `${this.baseURL}/chat/completions`,
        {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            Authorization: `Bearer ${this.apiKey}`,
          },
          body: JSON.stringify({
            model,
            messages,
            max_tokens: 500,
            temperature: 0.7,
          }),
        }
      );

      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.error?.message || 'API Error');
      }

      const data = await response.json();
      return data.choices[0].message.content;
    } catch (error) {
      console.error('OpenAI error:', error);
      throw error;
    }
  }

  async transcribeAudio(audioUri: string): Promise<string> {
    const formData = new FormData();
    formData.append('file', {
      uri: audioUri,
      type: 'audio/wav',
      name: 'audio.wav',
    } as any);
    formData.append('model', 'whisper-1');

    const response = await fetch(
      `${this.baseURL}/audio/transcriptions`,
      {
        method: 'POST',
        headers: {
          Authorization: `Bearer ${this.apiKey}`,
        },
        body: formData,
      }
    );

    const data = await response.json();
    return data.text;
  }

  async generateImage(prompt: string): Promise<string> {
    const response = await fetch(
      `${this.baseURL}/images/generations`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${this.apiKey}`,
        },
        body: JSON.stringify({
          prompt,
          model: 'dall-e-3',
          n: 1,
          size: '1024x1024',
        }),
      }
    );

    const data = await response.json();
    return data.data[0].url;
  }
}
```

```typescript
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const listRef = useRef<FlatList>(null);

  const openai = new OpenAIService(process.env.OPENAI_API_KEY || '');

  const handleSend = async () => {
    if (!input.trim()) return;

    const userMessage: ChatMessage = {
      role: 'user',
      content: input,
    };

    const newMessages = [...messages, userMessage];
    setMessages(newMessages);
    setInput('');
    setIsLoading(true);

    try {
      const response = await openai.chat(newMessages);
      setMessages((prev) => [
        ...prev,
        { role: 'assistant', content: response },
      ]);
    } catch (error) {
      console.error('Chat error:', error);
    } finally {
      setIsLoading(false);
    }
  };

  useEffect(() => {
    listRef.current?.scrollToEnd({ animated: true });
  }, [messages]);

  return (
    <View style={styles.container}>
      <FlatList
        ref={listRef}
        data={messages}
        keyExtractor={(_, idx) => idx.toString()}
        renderItem={({ item }) => (
          <View style={styles.messageContainer}>
            <View
              style={
                item.role === 'user'
                  ? styles.userBubble
                  : styles.assistantBubble
              }
            >
              <Text
                style={
                  item.role === 'user'
                    ? styles.userText
                    : styles.assistantText
                }
              >
                {item.content}
              </Text>
            </View>
          </View>
        )}
      />

      {isLoading && <ActivityIndicator size="small" color="#007AFF" />}

      <View style={styles.inputContainer}>
        <TextInput
          style={styles.textInput}
          placeholder="Ask anything..."
          value={input}
          onChangeText={setInput}
          editable={!isLoading}
          multiline
        />
        <TouchableOpacity
          style={styles.sendButton}
          onPress={handleSend}
          disabled={isLoading || !input.trim()}
        >
          <Text style={{ color: '#fff', fontSize: 16 }}>→</Text>
        </TouchableOpacity>
      </View>
    </View>
  );
}
```

```typescript
export class GeminiService {
  private baseURL = 'https://generativelanguage.googleapis.com/v1beta/models';

  constructor(private apiKey: string) {}

  async generateContent(prompt: string): Promise<string> {
    const response = await fetch(
      `${this.baseURL}/gemini-1.5-pro:generateContent?key=${this.apiKey}`,

        body: JSON.stringify({
          contents: [
            {
              parts: [{ text: prompt }],
            },
          ],
        }),
      }
    );

    const data = await response.json();
    return data.candidates[0].content.parts[0].text;
  }

  async generateImageCaptions(imageUri: string): Promise<string> {
    const response = await fetch(
      `${this.baseURL}/gemini-1.5-flash:generateContent?key=${this.apiKey}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [
            {
              parts: [
                { text: 'Describe this image in detail:' },
                {
                  inlineData: {
                    mimeType: 'image/jpeg',
                    data: imageUri, // base64 encoded
                  },
                },
              ],
            },
          ],
        }),
      }
    );

    const data = await response.json();
    return data.candidates[0].content.parts[0].text;
  }
}
```

## Recommendation Engine with Collaborative Filtering

**src/utils/recommendations.ts**
```typescript
export interface Item {
  id: string;
  title: string;
  category: string;
  embedding: number[];
}

export function getUserRecommendations(
  userInteractions: { itemId: string; rating: number }[],
  allItems: Item[],
  topN = 5
): Item[] {
  // Calculate user preference vector (simplified)
  const userVector = calculateUserVector(userInteractions, allItems);

  // Score items by similarity to user preferences
  const scored = allItems.map((item) => ({
    ...item,
    score: cosineSimilarity(userVector, item.embedding),
  }));

  // Filter already interacted items
  const recommendedItemIds = new Set(
    userInteractions.map((ui) => ui.itemId)
  );

  return scored
    .filter((item) => !recommendedItemIds.has(item.id))
    .sort((a, b) => b.score - a.score)
    .slice(0, topN);
}

function calculateUserVector(
  interactions: { itemId: string; rating: number }[],
  allItems: Item[]
): number[] {
  const itemMap = new Map(allItems.map((i) => [i.id, i]));
  const vectors = interactions
    .map((i) => itemMap.get(i.itemId)?.embedding || [])
    .filter((v) => v.length > 0);

  if (vectors.length === 0) return Array(100).fill(0);

  const dim = vectors[0].length;
  const result = Array(dim).fill(0);

  for (const vector of vectors) {
    for (let i = 0; i < dim; i++) {
      result[i] += vector[i] / vectors.length;
    }
  }

  return result;
}

function cosineSimilarity(a: number[], b: number[]): number {
  const dot = a.reduce((sum, val, i) => sum + val * b[i], 0);
  const normA = Math.sqrt(a.reduce((sum, val) => sum + val * val, 0));
  const normB = Math.sqrt(b.reduce((sum, val) => sum + val * val, 0));
  return normA && normB ? dot / (normA * normB) : 0;
}
```

## Summary

### On-Device ML Strengths
- ✓ Privacy-first
- ✓ Offline capability
- ✓ Low latency
- ✗ Large bundle
- ✗ Limited accuracy

### Cloud AI Strengths
- ✓ High accuracy
- ✓ Continuous improvement
- ✓ Smaller app size
- ✗ Requires internet
- ✗ API costs

**Best Practice**: Combine both. Use on-device ML for fast, privacy-critical tasks. Use cloud APIs for high-accuracy, complex reasoning. This hybrid approach delivers the best user experience.
