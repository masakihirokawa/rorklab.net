export default function SubscriptionWidget() {

  const [loading, setLoading] = useState(false);

    setLoading(true);
    try {
      // [1] Request checkout session from backend
      const response = await fetch("/api/create-checkout-session", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({

          userId: getCurrentUserId(),
          successUrl: `${window.location.origin}/subscribe/success`,
          cancelUrl: `${window.location.origin}/subscribe/cancel`
        })
      });

        process.env.NEXT_PUBLIC_STRIPE_PUBLISHABLE_KEY!

      console.error("Checkout failed:", error);
      alert("Checkout failed. Please try again.");
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="subscription-plans">
      {PLANS.map((plan) => (
        <div key={plan.id} className="plan-card">
          <h3>{plan.name}</h3>
          <div className="price">${(plan.price / 100).toFixed(2)}</div>
          <div className="period">
            {plan.interval === "month"
              ? "/mo"
              : plan.interval === "year"
              ? "/year"
              : "(one-time)"}
          </div>
          <ul className="features">
            {plan.features.map((feat) => (
              <li key={feat}>✓ {feat}</li>
            ))}
          </ul>
          <button
            onClick={() => handleCheckout(plan)}
            disabled={loading}
            className="cta-button"
          >
            {loading ? "Processing..." : "Get Started"}
          </button>
        </div>
      ))}
    </div>
  );

## Step 3: Backend Implementation — Stripe API Integration

### 3-1: Create Checkout Session

```typescript
// src/pages/api/create-checkout-session.ts
import Stripe from "stripe";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: "2024-04-10"
});

export default async function handler(req: any, res: any) {
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Method not allowed" });
  }

  const { planId, userId, successUrl, cancelUrl } = req.body;

  // Validate plan
  const plan = PLANS.find((p) => p.id === planId);
  if (!plan) {
    return res.status(404).json({ error: "Plan not found" });
  }

  try {
    // Create Stripe Checkout session
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ["card"],
      line_items: [
        {
          price: plan.stripePriceId,
          quantity: 1
        }
      ],
      mode:
        plan.interval === "one-time"
          ? "payment"        // One-time purchase
          : "subscription",  // Recurring subscription
      customer_email: getUserEmail(userId),
      metadata: {
        userId,
        planId,
        appName: "YourAppName"
      },
      success_url: successUrl + "?session_id={CHECKOUT_SESSION_ID}",
      cancel_url: cancelUrl,
      // Add trial for subscription plans
      subscription_data: plan.trialDays && plan.interval !== "one-time" ? {
        trial_period_days: plan.trialDays
      } : undefined
    });

    res.status(200).json({ sessionId: session.id });
  } catch (error) {
    console.error("Stripe error:", error);
    res.status(500).json({ error: "Failed to create checkout session" });
  }
}
```

### 3-2: Webhook Handler — Auto-Process Subscriptions

```typescript
// src/pages/api/webhook/stripe.ts
import { buffer } from "micro";
import Stripe from "stripe";
import { updateUserSubscription, revokeSubscription } from "@/lib/db";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET!;

export const config = {
  api: { bodyParser: { raw: true } }
};

export default async function handler(req: any, res: any) {
  if (req.method !== "POST") {
    return res.status(405).end();
  }

  const buf = await buffer(req);
  const sig = req.headers["stripe-signature"];

  let event: Stripe.Event;

  try {
    event = stripe.webhooks.constructEvent(buf.toString(), sig, webhookSecret);
  } catch (error) {
    console.error("Webhook verification failed:", error);
    return res.status(400).send("Webhook Error");
  }

  // Handle important events
  switch (event.type) {
    case "customer.subscription.updated": {
      const subscription = event.data.object as Stripe.Subscription;
      const userId = subscription.metadata.userId;

      // Save to KV or database
      await updateUserSubscription(userId, {
        stripeCustomerId: subscription.customer as string,
        subscriptionId: subscription.id,
        planId: subscription.metadata.planId,
        status: subscription.status,
        currentPeriodEnd: new Date(
          subscription.current_period_end * 1000
        )
      });

      console.log(`✓ Subscription updated for user ${userId}`);
      break;
    }

    case "customer.subscription.deleted": {
      const subscription = event.data.object as Stripe.Subscription;
      const userId = subscription.metadata.userId;

      // Revoke premium access
      await revokeSubscription(userId);
      console.log(`✗ Subscription cancelled for user ${userId}`);
      break;
    }

    case "invoice.payment_succeeded": {
      const invoice = event.data.object as Stripe.Invoice;
      console.log(`✓ Payment successful: ${invoice.id}`);
      // Send receipt email, etc.
      break;
    }

    case "invoice.payment_failed": {
      const invoice = event.data.object as Stripe.Invoice;
      console.log(`✗ Payment failed: ${invoice.id}`);
      // Send retry notification
      break;
    }
  }

  res.status(200).json({ received: true });
}
```

### 5-2: Trial Logic Implementation

```typescript
// src/lib/subscription-helpers.ts

export async function hasActivePremium(userId: string): Promise<boolean> {
  const sub = await KV.get(`sub:${userId}`);
  
  if (!sub) return false;

  const { currentPeriodEnd, status } = JSON.parse(sub);
  
  // Only "active" or "past_due" count as valid
  if (!["active", "past_due"].includes(status)) {
    return false;
  }

  return new Date() < new Date(currentPeriodEnd);
}

export async function getTrialDaysRemaining(
  userId: string
): Promise<number> {
  const trialStart = await KV.get(`trial:${userId}`);
  
  if (!trialStart) {
    // User never started trial → 7 days available
    return 7;
  }

  const daysPassed = Math.floor(
    (Date.now() - parseInt(trialStart)) / (1000 * 60 * 60 * 24)
  );
  
  return Math.max(0, 7 - daysPassed);
}

export async function startTrial(userId: string): Promise<void> {
  await KV.put(`trial:${userId}`, Date.now().toString());
}
```

## Step 7: Troubleshooting & Common Issues

### Q1: App Store rejected for "misleading billing"

**A**: Check:
1. Use App Store in-app purchasing (Stripe direct = rejection)
2. Verify price/cycle match between app and App Store Console exactly
3. Ensure cancellation works from iOS Settings → Subscriptions (one tap)

Rork apps must use **StoreKit2 wrapper** to bridge Apple's native IAP with Stripe.

### Q2: User complains about unexpected auto-renewal charge

**A**: This signals **UI clarity failure**. Fix:
1. Before purchase: Show "TAPPING = AUTO-RENEWS" in **red text**
2. 3 days before trial ends: Send in-app notification
3. Just before first charge: Send confirmation email

This 3-step system reduces billing complaints by 80%.

### Q3: Stripe fees are eating profits

**A**: Enable **local bank transfer payments** (where available). Stripe Japan supports ginkō furikomi (銀行振込) with 3.6% fee vs. 3.9% for card payments—small but cumulative savings.

```typescript

```
