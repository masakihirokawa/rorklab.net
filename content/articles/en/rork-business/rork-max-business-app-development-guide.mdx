## Introduction

Building business applications requires far more than UI design. User authentication, payment processing, push notifications, analytics, and customer support systems are essential for production-grade apps.

This article covers the complete development lifecycle of a business app using Rork Max, from initial planning through App Store and Google Play publication. The intended audience is startups targeting 50,000+ monthly active users.

## Defining Business Requirements

### Phase 0: Product Specification Document

Before starting development in Rork Max, create these requirement documents.

**1. User Segmentation**
```
[Target User Definition]
- Primary: 25-45 year old business professionals, willing to pay $9.99/month
```

## UI Implementation and Authentication Integration

### Step 1: User Authentication Flow

```swift
// Authentication flow generated by Rork Max (Swift/iOS)
import SwiftUI
import AuthenticationServices

@MainActor
final class AuthViewModel: ObservableObject {
    @Published var isAuthenticated = false
    @Published var currentUser: User?
    @Published var authError: AuthError?

    private let authService: AuthService
    private let keychainService: KeychainService

    init() {
        self.authService = AuthService()
        self.keychainService = KeychainService()
        checkAuthStatus()
    }

    func checkAuthStatus() {
        // Retrieve token from local keychain
        if let token = keychainService.retrieveToken() {
            validateToken(token)
        }
    }

    @MainActor
    func signInWithEmail(_ email: String, password: String) async {
        do {
            let response = try await authService.signIn(
                email: email,
                password: password
            )

            // Store token in Keychain
            try keychainService.storeToken(response.accessToken)
            self.currentUser = response.user
            self.isAuthenticated = true
        } catch {
            self.authError = AuthError.loginFailed
        }
    }

    @MainActor
    func signUpWithEmail(_ email: String, password: String, name: String) async {
        do {
            let response = try await authService.signUp(
                email: email,
                password: password,
                name: name
            )

            try keychainService.storeToken(response.accessToken)
            self.currentUser = response.user
            self.isAuthenticated = true
        } catch {
            self.authError = AuthError.signupFailed
        }
    }

    @MainActor
    func signOut() {
        keychainService.deleteToken()
        self.isAuthenticated = false
        self.currentUser = nil
    }

    private func validateToken(_ token: String) {
        Task {
            do {
                let user = try await authService.validateToken(token)
                self.currentUser = user
                self.isAuthenticated = true
            } catch {
                keychainService.deleteToken()
                self.isAuthenticated = false
            }
        }
    }
}

// Backend (Node.js + Express)
app.post('/api/auth/signup', async (req, res) => {
    const { email, password, name } = req.body;

    try {
        // Hash password
        const hashedPassword = await bcrypt.hash(password, 10);

        // Save user to database
        const user = await User.create({
            email,
            password: hashedPassword,
            name,
        });

        // Generate JWT token
        const token = jwt.sign(
            { userId: user.id, email: user.email },
            process.env.JWT_SECRET,
            { expiresIn: '7d' }
        );

        res.json({
            accessToken: token,
            user: {
                id: user.id,
                email: user.email,
                name: user.name,
            },
        });
    } catch (error) {
        res.status(400).json({ error: 'Signup failed' });
    }
});
```

### Step 4: Stripe Integration for Payment Processing

```swift
// Stripe payment flow (Swift)
import Stripe

@MainActor
class SubscriptionViewModel: ObservableObject {
    @Published var subscriptionStatus: SubscriptionStatus?
    @Published var isProcessing = false

    private let stripeService: StripeService

    init() {
        self.stripeService = StripeService()
    }

    @MainActor
    func startSubscription(priceID: String) async {
        isProcessing = true
        defer { isProcessing = false }

        do {
            // Step 1: Present payment sheet
            let result = try await stripeService.presentPaymentSheet(
                priceID: priceID
            )

            guard let paymentResult = result else { return }

            // Step 2: Confirm with backend
            try await confirmSubscription(
                paymentMethodID: paymentResult.paymentMethodId
            )

            // Step 3: Update user state
            self.subscriptionStatus = .active
        } catch {
            print("Subscription failed: \(error)")
        }
    }

    private func confirmSubscription(paymentMethodID: String) async throws {
        let response = try await URLSession.shared.data(
            from: URL(string: "https://your-api.com/api/subscribe")!,
            method: "POST",
            body: [
                "paymentMethodId": paymentMethodID,
            ]
        )

        if let data = response.0 as? Data {
            let decodedResponse = try JSONDecoder().decode(
                SubscriptionResponse.self,
                from: data
            )
            // Save token, update UI
        }
    }
}

// Backend (Node.js + Express)
const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);

app.post('/api/subscribe', async (req, res) => {
    const { userId, priceId, paymentMethodId } = req.body;

    try {
        // Step 1: Create/retrieve customer
        let customer = await stripe.customers.create({
            email: req.user.email,
            payment_method: paymentMethodId,
            invoice_settings: {
                default_payment_method: paymentMethodId,
            },
        });

        // Step 2: Create subscription
        const subscription = await stripe.subscriptions.create({
            customer: customer.id,
            items: [{ price: priceId }],
            expand: ['latest_invoice.payment_intent'],
        });

        // Step 3: Save to database
        await User.findByIdAndUpdate(userId, {
            stripeCustomerId: customer.id,
            subscriptionStatus: 'active',
        });

        res.json({ subscriptionId: subscription.id, status: 'success' });
    } catch (error) {
        res.status(400).json({ error: error.message });
    }
});
```

### Step 5: APNs (Apple Push Notification Service) Setup

```swift
// iOS push notifications implementation
import UserNotifications

class PushNotificationManager: NSObject, UNUserNotificationCenterDelegate {
    static let shared = PushNotificationManager()

    override init() {
        super.init()
        UNUserNotificationCenter.current().delegate = self
    }

    func requestUserPermission() {
        UNUserNotificationCenter.current().requestAuthorization(options: [.alert, .sound, .badge]) { granted, error in
            if granted {
                DispatchQueue.main.async {
                    UIApplication.shared.registerForRemoteNotifications()
                }
            }
        }
    }

    // Called from AppDelegate
    func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
        UNUserNotificationCenter.current().delegate = self
        return true
    }

    func userNotificationCenter(
        _ center: UNUserNotificationCenter,
        didReceive response: UNNotificationResponse,
        withCompletionHandler completionHandler: @escaping () -> Void
    ) {
        let userInfo = response.notification.request.content.userInfo

        if let notificationType = userInfo["type"] as? String {
            switch notificationType {
            case "message":
                handleChatNotification(userInfo)
            case "payment":
                handlePaymentNotification(userInfo)
            default:
                break
            }
        }

        completionHandler()
    }

    private func handleChatNotification(_ userInfo: [AnyHashable: Any]) {
        NotificationCenter.default.post(name: NSNotification.Name("NavigateToChatNotification"), object: userInfo)
    }

    private func handlePaymentNotification(_ userInfo: [AnyHashable: Any]) {
        print("Payment notification received")
    }
}
```

### Step 6: Firebase Cloud Messaging Integration

```swift
// FCM token retrieval and registration
import FirebaseMessaging

class FCMManager: NSObject, MessagingDelegate {
    static let shared = FCMManager()

    func setupFCM() {
        Messaging.messaging().delegate = self

        // Retrieve FCM token
        Messaging.messaging().token { token, error in
            if let error = error {
                print("FCM token error: \(error)")
            } else if let token = token {
                print("FCM Token: \(token)")
                self.registerFCMToken(token)
            }
        }
    }

    private func registerFCMToken(_ token: String) {
        Task {
            do {
                var request = URLRequest(url: URL(string: "https://your-api.com/api/register-fcm-token")!)
                request.httpMethod = "POST"
                request.setValue("application/json", forHTTPHeaderField: "Content-Type")
                request.httpBody = try JSONEncoder().encode(["fcmToken": token])

                let (_, response) = try await URLSession.shared.data(for: request)
                if let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 {
                    print("FCM token registered successfully")
                }
            } catch {
                print("Failed to register FCM token: \(error)")
            }
        }
    }

    func messaging(_ messaging: Messaging, didReceiveRegistrationToken fcmToken: String?) {
        if let token = fcmToken {
            registerFCMToken(token)
        }
    }
}
```

## Frequently Asked Questions

### Q1: What's the timeline from MVP to production?

**A:** 3-4 months with proper planning. One month for specification, 8 weeks for development/testing, 2 weeks for review/optimization. Rork Max can compress development by 4 weeks.

### Q2: How does the app scale to 10 million users?

**A:** Backend scalability is critical. Supabase auto-scales as a PaaS. Firebase charges by usage. However, Firebase Realtime caps at ~100,000 concurrent connections—consider migrating to a WebSocket server (Socket.io) at larger scales.

### Q3: What about security for payment processing?

**A:** Stripe achieves PCI Level 1 compliance. Implement client-side code that never touches card numbers (use Payment Method IDs instead). Backend must store API keys in environment variables, never in source code.

### Q4: App Store rejection—how to respond?

**A:** Check rejection reasons and resubmit with fixes. Common issues:
- Incomplete onboarding → Add post-sign-up quick tour
- Metadata inaccuracy → Ensure screenshots match app behavior
- Payment clarity → Clearly show pricing and cancellation process

### Q5: What's the post-launch marketing strategy?

**A:** Focus on App Store Optimization (ASO):
- Keyword optimization
- User review management
- A/B test screenshots and descriptions
- Apple Search Ads for paid acquisition
- Product Hunt/Reddit launches
