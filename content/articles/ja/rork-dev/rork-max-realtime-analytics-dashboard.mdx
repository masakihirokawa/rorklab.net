## この記事で学べること

アプリの成長を把握するには、ユーザーの行動をリアルタイムで可視化できるダッシュボードが欠かせません。この記事では、Rork MaxのネイティブSwiftコード生成能力とSupabase Realtimeを組み合わせて、WebSocketベースのリアルタイム分析ダッシュボードを一から構築する方法を解説します。

完成すると、以下の機能を持つダッシュボードアプリが手に入ります。

- アクティブユーザー数のリアルタイム表示
- イベントストリームの自動更新フィード
- 時系列チャートによるトレンド可視化
- フィルタリングとドリルダウン機能

対象読者はRork Maxの基本操作を理解しており、Supabaseの初期設定が完了している方です。

### Supabaseプロジェクトのセットアップ

まず、Supabase側にイベントデータを格納するテーブルを作成します。Supabase DashboardのSQL Editorで以下を実行してください。

```sql
-- アナリティクスイベントテーブル
create table analytics_events (
  id uuid default gen_random_uuid() primary key,
  event_name text not null,
  user_id uuid,
  properties jsonb default '{}',
  created_at timestamptz default now()
);

-- リアルタイム配信を有効化
alter publication supabase_realtime add table analytics_events;

-- 時系列集計用のビュー
create or replace view hourly_event_counts as
select
  date_trunc('hour', created_at) as hour,
  event_name,
  count(*) as event_count
from analytics_events
where created_at > now() - interval '24 hours'
group by date_trunc('hour', created_at), event_name
order by hour desc;

-- RLSポリシー（管理者のみ閲覧可能）
alter table analytics_events enable row level security;
create policy "Admin read access"
  on analytics_events for select
  using (auth.jwt() ->> 'role' = 'admin');
```

このスキーマのポイントは、`properties`カラムをJSONB型にしている点です。画面名、ボタンID、バージョン番号など、任意のメタデータを柔軟に格納できます。

## WebSocket接続の実装

Rork Maxが生成するSwiftコードに、Supabase Realtimeクライアントを統合します。以下はコアとなる`RealtimeManager`クラスの実装です。

```swift
import Foundation
import Supabase

// リアルタイムイベントを管理するクラス
@MainActor
class RealtimeManager: ObservableObject {
    @Published var activeUsers: Int = 0
    @Published var recentEvents: [AnalyticsEvent] = []
    @Published var isConnected: Bool = false

    private let client: SupabaseClient
    private var channel: RealtimeChannelV2?

    init() {
        // Supabaseクライアントの初期化
        self.client = SupabaseClient(
            supabaseURL: URL(string: "https://your-project.supabase.co")!,
            supabaseKey: "your-anon-key"
        )
    }

    // WebSocketチャネルの購読を開始
    func subscribe() async {
        let channel = client.realtimeV2.channel("analytics")

        // INSERTイベントを監視
        let insertions = channel.postgresChange(
            InsertAction.self,
            table: "analytics_events"
        )

        await channel.subscribe()
        self.channel = channel
        self.isConnected = true

        // イベントストリームを非同期で処理
        for await insertion in insertions {
            let event = AnalyticsEvent(
                id: insertion.record["id"]?.stringValue ?? "",
                eventName: insertion.record["event_name"]?.stringValue ?? "",
                properties: insertion.record["properties"]?.objectValue ?? [:],
                createdAt: Date()
            )
            // 最新50件を保持
            self.recentEvents.insert(event, at: 0)
            if self.recentEvents.count > 50 {
                self.recentEvents.removeLast()
            }
            // アクティブユーザー数の再計算
            await self.refreshActiveUsers()
        }
    }

    // 直近5分間のユニークユーザー数を取得
    func refreshActiveUsers() async {
        do {
            let response = try await client
                .from("analytics_events")
                .select("user_id")
                .gte("created_at", value: ISO8601DateFormatter()
                    .string(from: Date().addingTimeInterval(-300)))
                .execute()
            // ユニークなuser_idをカウント
            let decoded = try JSONDecoder()
                .decode([[String: String]].self, from: response.data)
            let uniqueUsers = Set(decoded.compactMap { $0["user_id"] })
            self.activeUsers = uniqueUsers.count
        } catch {
            print("Active users fetch error: \(error)")
        }
    }

    // 購読を解除
    func unsubscribe() async {
        if let channel = self.channel {
            await client.realtimeV2.removeChannel(channel)
        }
        self.isConnected = false
    }
}
```

このコードの重要なポイントは3つあります。

1つ目は`@MainActor`アノテーションです。UIの更新はメインスレッドで行う必要があるため、クラス全体をメインアクターに配置しています。

2つ目は`for await`ループによるストリーム処理です。Supabase Swift SDKのRealtime V2は`AsyncSequence`に準拠しており、Swift Concurrencyと自然に統合できます。

3つ目はバッファリング戦略です。最新50件のみを保持することで、長時間稼働してもメモリ使用量が増加しません。

## バッチ処理とパフォーマンス最適化

リアルタイム配信は便利ですが、高トラフィックな環境ではWebSocket接続がボトルネックになる可能性があります。以下の最適化テクニックを適用しましょう。

### イベントバッチング

個々のイベントを逐一送信する代わりに、一定時間ごとにまとめて送信する仕組みを実装します。

```swift
actor EventBatcher {
    private var buffer: [[String: AnyJSON]] = []
    private let client: SupabaseClient
    private let flushInterval: TimeInterval = 5.0
    private let maxBatchSize: Int = 50

    init(client: SupabaseClient) {
        self.client = client
        Task { await self.startPeriodicFlush() }
    }

    func add(event: [String: AnyJSON]) {
        buffer.append(event)
        if buffer.count >= maxBatchSize {
            Task { await flush() }
        }
    }

    private func flush() async {
        guard !buffer.isEmpty else { return }
        let batch = buffer
        buffer = []
        do {
            try await client
                .from("analytics_events")
                .insert(batch)
                .execute()
        } catch {
            // 失敗時はバッファに戻す
            buffer.insert(contentsOf: batch, at: 0)
            print("Batch insert failed: \(error)")
        }
    }

    private func startPeriodicFlush() async {
        while !Task.isCancelled {
            try? await Task.sleep(for: .seconds(flushInterval))
            await flush()
        }
    }
}
// 期待する出力: 5秒ごとまたは50件ごとにバッチINSERTが実行される
```

### ダッシュボード側のスロットリング

大量のイベントがリアルタイムで流れてくる場合、UIの更新頻度を制限する必要があります。

```swift
extension RealtimeManager {
    // UIの更新を最大1秒に1回に制限
    func throttledRefresh() {
        let now = Date()
        if now.timeIntervalSince(lastRefresh) < 1.0 {
            return
        }
        lastRefresh = now
        Task { await refreshActiveUsers() }
    }
}
```

## セキュリティとアクセス制御

分析ダッシュボードには機密データが含まれるため、適切なアクセス制御が不可欠です。

### Row Level Security（RLS）の強化

```sql
-- 管理者ロールのみダッシュボードデータにアクセス可能
create policy "Dashboard admin only"
  on analytics_events for select
  using (
    exists (
      select 1 from user_roles
      where user_roles.user_id = auth.uid()
      and user_roles.role = 'admin'
    )
  );

-- イベント書き込みは認証済みユーザーのみ
create policy "Authenticated insert"
  on analytics_events for insert
  with check (auth.uid() is not null);
```

### APIキーの安全な管理

Rork Maxで生成されたプロジェクトでは、APIキーをハードコードせずに環境設定から読み込むようにします。

```swift
enum Config {
```

## よくある質問

### Q1: Supabase Realtimeの無料プランで十分ですか？

Supabaseの無料プランでは同時接続数200まで、月間のRealtime メッセージ数200万件までサポートされています。個人開発やプロトタイプには十分ですが、本番環境で多数のユーザーを処理する場合はProプラン（$25/月）へのアップグレードを検討してください。Realtime接続数は管理画面の「Realtime Inspector」で監視できます。

### Q2: WebSocket接続が切断された場合の復旧はどうしますか？

Supabase Swift SDKには自動再接続機能が組み込まれています。ただし、長時間の切断後はデータの欠損が発生する可能性があります。`RealtimeManager`に再接続検出のロジックを追加し、接続復帰時にHTTPで差分データを取得する「ギャップフィル」パターンを実装するのがベストプラクティスです。

### Q3: React NativeベースのRorkでも同じことができますか？

基本的な構成は同じですが、UIコンポーネントはReact Native用のライブラリ（react-native-charts-wrapper等）を使う必要があります。また、WebSocket管理にはReact Nativeの制約（バックグラウンド時の接続維持など）を考慮する必要があるため、Rork MaxのネイティブSwiftアプローチの方が信頼性と性能の面で優位です。

### Q4: ダッシュボードをiPadやMacにも対応させるには？

Rork MaxはSwiftUIベースのコードを生成するため、iPad対応は比較的簡単です。`NavigationSplitView`を使ってサイドバーレイアウトに切り替え、`horizontalSizeClass`環境変数で画面サイズに応じたレイアウト分岐を行います。Mac Catalyst対応もSwiftUIなら最小限の変更で実現できます。
