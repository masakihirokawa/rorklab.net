```typescript
// src/pages/api/create-checkout-session.ts
import Stripe from "stripe";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: "2024-04-10"
});

export default async function handler(req: any, res: any) {
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Method not allowed" });
  }

  const { planId, userId, successUrl, cancelUrl } = req.body;

  // プラン情報を取得
  const plan = PLANS.find((p) => p.id === planId);
  if (!plan) {
    return res.status(404).json({ error: "Plan not found" });
  }

  try {
    // Stripe Checkout Session 作成
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ["card"],
      line_items: [
        {
          price: plan.stripePriceId,
          quantity: 1
        }
      ],
      mode:
        plan.interval === "one-time"
          ? "payment"        // 買い切り
          : "subscription",  // サブスク
      customer_email: getUserEmail(userId), // ユーザーメール
      metadata: {
        userId,
        planId,
        appName: "YourAppName"
      },
      success_url: successUrl + "?session_id={CHECKOUT_SESSION_ID}",
      cancel_url: cancelUrl
    });

    res.status(200).json({ sessionId: session.id });
  } catch (error) {
    console.error("Stripe error:", error);
    res.status(500).json({ error: "Checkout creation failed" });
  }
}
```

### 3-2: Webhook ハンドラ——サブスク更新の自動処理

```typescript
// src/pages/api/webhook/stripe.ts
import { buffer } from "micro";
import Stripe from "stripe";
import { updateUserSubscription, revokeSubscription } from "@/lib/db";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET!;

export const config = {
  api: { bodyParser: { raw: true } }
};

export default async function handler(req: any, res: any) {
  if (req.method !== "POST") {
    return res.status(405).end();
  }

  const buf = await buffer(req);
  const sig = req.headers["stripe-signature"];

  let event: Stripe.Event;

  try {
    event = stripe.webhooks.constructEvent(buf.toString(), sig, webhookSecret);
  } catch (error) {
    console.error("Webhook signature verification failed:", error);
    return res.status(400).send(`Webhook Error`);
  }

  // 重要なイベントをハンドル
  switch (event.type) {
    case "customer.subscription.updated": {
      const subscription = event.data.object as Stripe.Subscription;
      const userId = subscription.metadata.userId;

      // KV/DB に権限を保存
      await updateUserSubscription(userId, {
        stripeCustomerId: subscription.customer as string,
        subscriptionId: subscription.id,
        planId: subscription.metadata.planId,
        status: subscription.status,
        currentPeriodEnd: new Date(subscription.current_period_end * 1000)
      });

      console.log(`✓ Subscription updated for user ${userId}`);
      break;
    }

    case "customer.subscription.deleted": {
      const subscription = event.data.object as Stripe.Subscription;
      const userId = subscription.metadata.userId;

      // 権限を削除
      await revokeSubscription(userId);
      console.log(`✗ Subscription deleted for user ${userId}`);
      break;
    }

    case "invoice.payment_succeeded": {
      const invoice = event.data.object as Stripe.Invoice;
      console.log(`✓ Payment succeeded: ${invoice.id}`);
      // 必要に応じて領収書メール送信など
      break;
    }

    case "invoice.payment_failed": {
      const invoice = event.data.object as Stripe.Invoice;
      console.log(`✗ Payment failed: ${invoice.id}`);
      // ユーザーに再試行促す通知
      break;
    }
  }

  res.status(200).json({ received: true });
}
```

### 5-2: トライアル自動付与ロジック

```typescript
// src/lib/subscription.ts

export async function hasActiveSubscription(userId: string): Promise<boolean> {
  // KV from Cloudflare Workers
  const subscription = await KV.get(`sub:${userId}`);
  
  if (!subscription) {
    return false;
  }

  const { currentPeriodEnd, status } = JSON.parse(subscription);
  
  // active | past_due | canceled | ... 
  if (status !== "active" && status !== "past_due") {
    return false;
  }

  // 有効期限チェック
  return new Date() < new Date(currentPeriodEnd);
}

export async function getTrialRemainingDays(userId: string): Promise<number> {
  const trialStart = await KV.get(`trial:${userId}`);
  
  if (!trialStart) {
    return 7; // まだトライアル未使用 → 7日利用可
  }

  const elapsed = Math.floor(
    (Date.now() - parseInt(trialStart)) / (1000 * 60 * 60 * 24)
  );
  const remaining = Math.max(0, 7 - elapsed);
  
  return remaining;
}
```

### 6-2: LTV（顧客生涯価値）計算＆改善

```typescript
// 平均顧客生涯価値 (LTV)
const ARPU = 980;           // 平均月額収益 (ARPU)
const RETENTION = 0.92;     // 月次リテンション率 92%
const LIFETIME = 1 / (1 - RETENTION);  // 平均契約期間（月）

const LTV = ARPU * LIFETIME;

const CAC_TARGET = LTV / 3;  // 約4,083円以下
```

## Step 7: トラブルシューティング＆FAQ

### Q1: App Store で「不適切な課金方式」と言われました

**A**: 以下を確認してください：
1. アプリ内課金で必ずApp Store のプロダクトを使用している（Stripe直接決済は NG）
2. 価格・サイクルが App Store Console に正確に登録されている
3. キャンセルが「設定」から1タップで完了可能

Rorkの場合、**StoreKit2 wrapper** を使用して、 Apple の課金フレームワークとStripeを正しく連携させるのが必須です。

### Q2: 顧客から「なぜ自動更新されたのか」とクレームが来ました

**A**: これは **UI透明性不足** の典型。改善策：
1. 購入前に「購入ボタン押下 = 自動更新開始」を**赤いテキスト**で警告
2. 無料トライアル期間終了の**3日前に通知**を送る
3. トライアル終了後、サブスク初回請求の**3時間前に確認メール**を送る

この3段階で、クレーム80%削減が可能です。

### Q3: Stripe の為替手数料が高くて利益が圧迫されています

**A**: **国内銀行振込対応** に変更すべき。Stripe は2024年から日本での銀行振込対応を強化しており、国内決済なら手数料 3.6% に抑えられます。
